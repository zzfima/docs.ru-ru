```vb  
  custDS.Tables("OrdersTable").Columns("CustomerID"))

custDS.Tables("OrdersTable").Constraints.Add(custOrderFK)  
```  
  
```csharp  
ForeignKeyConstraint custOrderFK = new ForeignKeyConstraint("CustOrderFK",
```  
