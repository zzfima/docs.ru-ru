# Предупреждение компилятора (уровень 2) CS0467
[!INCLUDE[vs2017banner](../../../csharp/includes/vs2017banner.md)]

Неоднозначность между методом "метод" и "не метод", который методом не является.Используйте группу методов.  
  
 Наследованные члены из разных интерфейсов, которые имеют одинаковые сигнатуры, вызывают ошибку неоднозначности.  
  
## Пример  
 Следующий пример проводит к возникновению предупреждения CS0467.  
  
```c#  
// CS0467.cs  
interface IList   
{  
    int Count { get; set; }  
}  
  
interface ICounter  
{  
    void Count(int i);  
}  
  
interface IListCounter : IList, ICounter {}  
  
class Driver   
{  
    void Test(IListCounter x)  
    {  
        // The following line causes the warning. The assignment also  
        // causes an error because you can't assign a value to a method.  
        x.Count = 1;  
        x.Count(3);     
        // To resolve the warning, you can change the name of the method or   
        // the property.  
  
        // You also can disambiguate by specifying IList or ICounter.  
        ((IList)x).Count = 1;  
        ((ICounter)x).Count(3);  
    }  
  
    static void Main()   
    {  
    }  
}  
```
