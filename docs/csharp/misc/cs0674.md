 <span data-ttu-id="6a2bb-106">Следующий пример приводит к возникновению ошибки CS0674:</span><span class="sxs-lookup"><span data-stu-id="6a2bb-106">The following sample generates CS0674:</span></span>  
  
```csharp  
// CS0674.cs  
using System;  
public class MyClass   
{  
  
   public static void UseParams([ParamArray] int[] list)   // CS0674  
   // try the following line instead  
   // public static void UseParams(params int[] list)   
   {  
      for ( int i = 0 ; i < list.Length ; i++ )  
         Console.WriteLine(list[i]);  
      Console.WriteLine();  
   }  
  
   public static void Main()   
   {  
      UseParams(1, 2, 3);  
   }  
}  
```
