## <a name="example"></a><span data-ttu-id="be15b-106">Пример</span><span class="sxs-lookup"><span data-stu-id="be15b-106">Example</span></span>

public class Test  
{  
    [return:System.CLSCompliant(true)]  // CS3023

    {

    }  
}
