```  
// CS0307.cs  
class C  
{  
   public int P { get { return 1; } }  
   public static void Main()  
   {  
      C c = new C();  
      int p = c.P<int>();  // CS0307 – C.P is a property  
      // Try this instead  
      // int p = c.P;  
   }  
}  
```
